ques23)Write a C program to convert days into years, weeks and days
ans23)
#include <stdio.h>

int main()
{
    int days, years, weeks;

    printf("Enter days:");
    scanf("%d", &days);
    years = (days/365);  
    weeks = (days % 365)/7;
    days  = days-((years * 365) + (weeks * 7));
    printf("YEARS: %d",years);
    printf("WEEKS: %d",weeks);
    printf("DAYS: %d",days);

    return 0;
}
